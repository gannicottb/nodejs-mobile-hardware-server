CREATE TABLE readings (
	id MEDIUMINT NOT NULL AUTO_INCREMENT,	
	ts TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	lat DECIMAL(9,6),
	lon DECIMAL(9,6),
	co DECIMAL(5,2),
	pm DECIMAL(5,2),
	hum DECIMAL(5,2),
	temp DECIMAL(5,2),
	PRIMARY KEY(id)
);
